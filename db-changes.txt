Querion Database Changes
========================
DROP TABLE IF EXISTS query_history;

CREATE TABLE query_history (
    id BIGSERIAL PRIMARY KEY,
    request_id VARCHAR(64) NOT NULL UNIQUE,
    user_query TEXT NOT NULL,
    query_type VARCHAR(32) NOT NULL,
    generated_sql TEXT,
    generated_sql_params TEXT,
    execution_status VARCHAR(32),
    execution_time_ms BIGINT,
    total_duration_ms BIGINT,
    load_duration_ms BIGINT,
    prompt_eval_duration_ms BIGINT,
    eval_duration_ms BIGINT,
    result_row_count BIGINT,
    result_preview_json TEXT,
    error_message TEXT,
    model_name VARCHAR(255),
    api_url VARCHAR(255),
    prompt_text TEXT,
    llm_response_text TEXT,
    final_answer_prompt TEXT,
    final_answer_response_text TEXT,
    final_answer TEXT,
    key_insights TEXT,
    llm_call_count INTEGER DEFAULT 0,
    prompt_tokens INTEGER,
    response_tokens INTEGER,
    score DOUBLE PRECISION DEFAULT 0,
    feedback_count INTEGER DEFAULT 0,
    positive_feedback_count INTEGER DEFAULT 0,
    negative_feedback_count INTEGER DEFAULT 0,
    user_feedback BOOLEAN,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_query_history_created_at
    ON query_history (created_at DESC);

CREATE INDEX idx_query_history_query_type_created_at
    ON query_history (query_type, created_at DESC);

CREATE TABLE customer (
id BIGSERIAL PRIMARY KEY,
customer_code VARCHAR(30) UNIQUE NOT NULL,
first_name VARCHAR(100) NOT NULL,
last_name VARCHAR(100),
email VARCHAR(150) UNIQUE,
phone VARCHAR(20),
company_name VARCHAR(150),
active BOOLEAN DEFAULT TRUE,
credit_limit NUMERIC(12,2) DEFAULT 0,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
